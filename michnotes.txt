ButtonHandler

enum button type {button, lever} //can add more lol 
public bool type = 0,1;

if hand is on button hitbox:
	- button / lever highlight
	- state = can press

if state = can press AND action button pressed: 
	- if type button:
		state = activated
	
	LEVER STATE: List<bool> [decel, neutral, accel]
	- if type lever:
		if action button pressed AND up:
			- animation
			- set current leverstate to false
			- leverstate++ unless greater than 2;
			- set current (new) leverstate to true

		
		if action button pressed AND down:
			- animation
			- set current leverstate to false
			- leverstate-- unless less than 0;
			- set current (new) leverstate to true
